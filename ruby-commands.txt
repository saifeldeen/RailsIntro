RAKE:
rake is c's version of make :)

rake db:migrate:status

# creates schema from scratch
rake db:schema:load

# runs all of the migrate scripts
rake db:migrate

# drops the db
rake db:drop

# remove
rake db:migrate:down VERSION=20130417085602

# rake migrate
rake db:migrate
